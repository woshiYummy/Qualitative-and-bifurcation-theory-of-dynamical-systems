几种典型的动力系统:
\begin{align*}
  \dot{x} & = a x, & &\text{指数模型,} \\
  \dot{x} & = a x - b x^2, &&\\
  \dot{x} & = a x (1 - x), & &\text{Logistic 模型.}
\end{align*}
一般的, 对于系统 $\dot{x} = f(x)$, 可将 $f(x)$ 通过 Taylor 展开为,
\begin{equation*}
  f(x) = f(0) + a_1 x + a_2 x^2 + \cdots.
\end{equation*}
一般地取 $f(0) = 0$. 取一次项系数 $a_1$ 不为 0, 其余项系数均为 0 时, 则为指数模型. 若一次项系数 $a_1 \neq 0$, 二次项系数 $a_2 < 0$, 其余项系数均为 0, 则为 Logistic 模型.

\section{解的 (局部) 存在, 唯一性}

\paragraph{基础知识回顾:}
\begin{itemize}
  \item 向量的和、差、数乘;
  \item 内积 $<x, y> = \sum \limits^n_{i = 1} x_i y_i$;
  \item 范数 $\|x\| = \sum \limits^n_{i = 1} |x_i|$ 或 $\|x\| = \sqrt{<x, x>} = \sqrt{\sum \limits^n_{i = 1} x^2_i}$;
  \item 性质 $|<x,y>| \leq \|x\| \cdot \|y\|$, $\|x + y\| \leq \|x\| + \|y\|$;
  \item 向量函数的导数、积分 $\Leftrightarrow$ 分量的导数、积分;
  \item
    \begin{equation}
      \left\| \int^t_{t_0} f(s) \kD s \right\| \leq \left| \int^t_{t_0} \|f(s)\| \kD s\right|.
    \end{equation}
\end{itemize}

\begin{enumerate}
\item 几乎所有的非线性方程不可解;
\item 定性, 稳定性, 分析为主要手段;
\item 解题存在唯一性 (数值计算).
\end{enumerate}

\paragraph{方程 (系统):}

\begin{equation}\label{Eq:1.14}
  \dot{x} = f(t, x),\ x(t_0) = x^0.
\end{equation}

\begin{Thm}\label{Th:1.1}$\ $
  \begin{enumerate}
    \item $f(t, x) \in C^0(G)$, $G: |t - t_0| \leq a$, $\|x - x_0\| \leq b$;

    \item $f$ 关于 $x$, Lipschitz 连续, $\exists L > 0$, s.t. $\|f(t,x) - f(t,y)\| \leq L \|x - y\|$.

    \item 记 $M = \max \limits_{G} \|f(t,x)\|$, $h = \min \left( a,\frac{b}{M} \right)$, $|t - t_0| < h$.
  \end{enumerate}
\end{Thm}
\begin{proof}
  简要概述证明的五个步骤.
  \begin{asparaenum}
    \item 证明方程 \ref{Eq:1.14} 与积分方程 \ref{Eq:1.15} 等价.
        \begin{equation}\label{Eq:1.15}
          x = x^0 + \int^t_{t_0} f(s, x(s)) \kD s.
        \end{equation}

    \item 逐次逼近. 在 $|t - t_0| \leq h$ 上定义 $\{\Phi_k(t)\}$, $k = 2, 3, \dots,$
        \begin{align*}
          \Phi_0(t) & = x^0, \\
          \Phi_1(t) & = x^0 + \int^t_{t_0} f(s, \Phi_0(s)) \kD s, \\
           & \cdots, \\
          \Phi_k(t) & = x^0 + \int^t_{t_0} f(s, \Phi_{k - 1}(s)) \kD s.
        \end{align*}

    \item 证明 $\Phi_k(t)$ 在 $|t - t_0| \leq h$ 上一致收敛.

    \item $x = \Phi(t) = \lim \limits_{k \to \infty} \Phi_k(t)$ 为方程 \ref{Eq:1.15} 的解.

    \item 唯一性.
  \end{asparaenum}
\end{proof}

\begin{Note}\
  \begin{enumerate}
    \item $f(t, x)$ 连续 $\Rightarrow$ 存在性 (欧拉法).

    \item Lipschitz 连续 $\Rightarrow$ 唯一性.
  \end{enumerate}
\end{Note}

\begin{enumerate}
\item 由定理 \ref{Th:1.1}, 存在 $h$, 使得在 $|t - t_0| \leq h$ 上, 存在唯一的解 $x = \Phi_1(t)$.

\item 记 $t_1 = t_0 + h$, $x_1 = \Phi_1(t_1) = \Phi_1(t_0 + h)$, 考虑初值问题
    \begin{equation}\tag{\ref{Eq:1.16}$'$}\label{Eq:1.16'}
      \dot{x} = f(t_1,x),\ x(t_1) = x_1
    \end{equation}
    又由定理 \ref{Th:1.1} 存在 $h_1$, 使得在 $|t - t_1| \leq h_1$ 上, 方程 \ref{Eq:1.16'} 存在唯一解 $x = \Phi_2(t)$.
\item 分段表示
    \begin{equation}
      \Phi^*(t) =
      \begin{cases}
        \Phi_1(t),\ t\in [t_0 - h,t_0 + h]\\
        \Phi_2(t),\ t\in [t_0 + h,t_1 + h_1]
      \end{cases}
    \end{equation}
    是方程 \ref{Eq:1.16} 在 $t \in [t_0 - h, t_1 + h_1]$ 上的唯一解.
\end{enumerate}

\begin{Thm}\label{Th:1.2}
  $f(t,x)$ 在 $G$ 上连续, 关于 $x$, Lipschitz 连续 $\Rightarrow$ 方程 \ref{Eq:1.16} 的解可以延拓到 $G$ 的边界.
\end{Thm}
\begin{proof}\
\begin{asparaenum}
  \item 设最大存在区间为 $(\alpha, \beta)$, $\beta$ 有限. $x = \Phi(t_0)$, ($t_0 < \beta$) 位于闭区域 $G' \subset G$.

        \includegraphics{T1_8}

  \item 存在 $M$, 使得 $\|f(t, \Phi(t))\| \leq M$ (于 $G'$), 所以 $\forall t_1, t_2 \in (t_0, \beta)$,
        \begin{align*}
          \left\|\Phi(t_1) - \Phi(t_2)\right\| &  = \left\|\int^{t_1}_{t_0} f(s, \Phi_1(s)) \kD s - \int^{t_2}_{t_0} f(s, \Phi_1(s)) \kD s\right\|\\
          & = \left\|\int^{t_1}_{t_2} f(s, \Phi_1(s)) \kD s\right\|\\
          & \leq \left|\int^{t_1}_{t_2} \left\|f(s, \Phi_1(s))\right\|\right| \kD s\\
          & \leq M|t_1 - t_2|.
        \end{align*}
        这说明函数 $\Phi(t)$ 是一致连续的.

        $\Rightarrow\Phi(\beta - 0)$ 存在且 $(\beta, \Phi(\beta - 0)) \in G'$

        $\Rightarrow$ 仍可延拓, 矛盾!
  \end{asparaenum}
\end{proof}

\paragraph{解的整体存在性}
\begin{Cor}\label{Co:1.3}
  $f(t,x) \in C^0(\mathbb{R}^{n + 1})$ 关于 $x$, Lipschitz 连续, 方程 \ref{Eq:1.16} 的解有界 $\Rightarrow$ 解存在于 $( - \infty, + \infty)$.
\end{Cor}
理由: $\mathbb{R}^{n + 1}( = G)$ 边界为: $x \to \pm \infty$ or $t \to \pm \infty$.

\begin{Cor}
  自治系统
  \begin{equation}\label{Eq:1.19}
    \frac{\D \bf{x}}{\D t} = {\bf f(x)},\ {\bf x}(t_0) = {\bf x}^0
  \end{equation}
  ${\bf f} \in C^0(\mathbb{R}^n)$, ${\bf f}$ 是 Lipschitz 连续的, 解有界 $\Rightarrow$ 解的存在区间为 $( - \infty, + \infty)$.
\end{Cor}

\begin{Cor}
  ${\bf f(x)} \in C^0(\mathbb{R}^n)$, 有界 (+ Lipschitz 连续, 书上没有该条件) $\Rightarrow$ 方程 \ref{Eq:1.19} 解存在区间为 $( - \infty, + \infty)$.
\end{Cor}

$f(x)$ 连续, 有界但不 Lipschitz 连续的例子:
\begin{Exa}
  \begin{equation}
    f(x) =
    \begin{cases}
      \sin \sqrt{x}, & x\geq 0\\
      0, & x\leq 0
    \end{cases}
  \end{equation}
\end{Exa}
\begin{proof}
  反设 $f(x)$ 局部 Lipschitz, 在原点的闭区间有一致的 Lipschitz 常数 $L$.
  \begin{equation*}
    \begin{array}{l @{} l}
      \Rightarrow |\sin \sqrt{x_1} - \sin \sqrt{x_2}| & = \left|\frac{1}{2 \sqrt{\xi}} \cos \sqrt{\xi} \right| \cdot |x_1 - x_2|\\
      & \leq L |x_1 - x_2| \\
      \Rightarrow \cos \sqrt{\xi} \leq 2 L \sqrt{\xi} & \\
      \because x_1, x_2 \to 0,\ \xi \to 0 & \\
      \Rightarrow 1 \leq 0,
    \end{array}
  \end{equation*}
  矛盾!
\end{proof}

\section{解对初值及参数的连续依赖性}

Logistic 方程 $\dot{x} = x(r - x)$, $x(t_0) = t_0$, $x = x(t,t_0,x_0,r)$.

\begin{Thm}\label{Th:1.6}\
  \begin{enumerate}
    \item 初值问题
        \begin{equation}
          \dot{x} = f(t, x, \mu), \ x(t_0) = x^0.
        \end{equation}
    \item $f(t, x, \mu)$ 的定义域为
        \begin{equation}
          G:\ |t - t_0| \leq a,\ \|x - x_0\| \leq b,\ \|\mu - \mu_0\| \leq c.
        \end{equation}
    \item 连续, 局部 Lipschitz.
  \end{enumerate}
  则 $x = \Phi(t, u)$ 局部存在且关于 $\mu$ 连续. (即参数连续.)
\end{Thm}
\begin{proof}
  思路如下:
  \begin{equation*}
    \dot{x} = f(t, x, \mu),\ x(t_0) = x^0 \Leftrightarrow \Phi(t, \mu) = x_0 + \int^t_{t_0} f(s, \Phi(s, \mu), \mu) \kD s.
  \end{equation*}
  定义
  \begin{align*}
    \Phi_0(t) & = x_0,\\
    \Phi_1(t, \mu) & = x_0 + \int^{t}_{t_0} f(s, \Phi_0(s), \mu) \kD s,\\
    \Phi_2(t, \mu) & = x_0 + \int^{t}_{t_0} f(s, \Phi_1(s, \mu), \mu) \kD s,\\
    \vdots &
  \end{align*}
  $\{\Phi_n(t, \mu)\}$ 一致收敛, $|t - t_0| \leq h$, $\|\mu - \mu_0\| \leq c$.
  \begin{equation*}
    \Phi(t,\mu) = x_0 + \int^t_{t_0} f(s, \Phi(s, \mu), \mu) \kD s \Rightarrow \Phi(t, \mu)
  \end{equation*}
  为 $t$, $\mu$ 的连续函数.
\end{proof}

\begin{Thm}\
  \begin{enumerate}
    \item 初值问题
        \begin{equation}
          \dot{x} = f(t, x),\ x(t_0) = \eta.
        \end{equation}

    \item 区域 $G:\ |t - t_0| \leq a$, $\|x - x_0\| \leq b$.

    \item 当 $\eta:\ \|\eta - x_0\| \leq \frac{b}{2}$.
  \end{enumerate}
  $\Rightarrow$ 存在唯一的 $x = \Phi(t, \eta)$ 关于 $t$, $\eta$ 连续.
\end{Thm}
\begin{proof}\
\begin{enumerate}[方法I. ]
\item 证明思路:
    \begin{asparaenum}
    \item 变换 $x = z + y$,
        \begin{equation}\label{Th:1.24}
          \dot{z} = f(t, z + \eta)
        \end{equation}
        将对初值的连续依赖问题转化为对参数的连续依赖.
    \item $\forall z_1, z_2$
        \begin{align*}
          \|f(t, z_1 + \eta) - f(t, z_2 + \eta)\| & \leq L \|(z_1 + \eta) - (z_2 + \eta)\| \\
         &  = L \|z_1 - z_2\|
        \end{align*}
    \item 由 \ref{Th:1.6}, 方程 \ref{Th:1.24} 的解 $z = \Psi(t, \eta)$ 关于 $(t, \eta)$ 连续.

        $\Rightarrow$ 存在唯一的 $x = z + \eta = \Psi(t, \eta) + \eta$, 且满足连续依赖性.
    \end{asparaenum}

\item 证明思路:
    \begin{align*}
    \begin{split}
      \dot{x}_1 = f(t,x_1),\ x_1(t_0) = \eta_1\\
      \dot{x}_2 = f(t,x_2),\ x_2(t_0) = \eta_2
    \end{split}\\
    \begin{split}
      \Rightarrow & x_1(t) - x_1(t_0) = \int^t_{t_0} f(s, x_1(s)) \kD s\\
                  & x_2(t) - x_2(t_0) = \int^t_{t_0} f(s, x_2(s)) \kD s
    \end{split}\\
    \begin{split}
      \Rightarrow & \|x_1(t, t_0, \eta_1) - x_2(t, t_0, \eta_2)\| \\
                  & \leq \|\eta_1 - \eta_2\| + \left\|\int^t_{t_0} \left[f(s, x_1(s)) - f(s, x_2(s))\right] \kD s\right\|\\
                  & \leq \|\eta_1 - \eta_2\| + L \int^t_{t_0} \|x_1(s) - x_2(s)\| \kD s\\
                  & \leq \|\eta_1 - \eta_2\| e^{L (t - t_0)}.~\footnotemark
    \end{split}% &
    \end{align*}
  \end{enumerate}
\end{proof}
\footnotetext
{
Grownwell-Bellman 不等式: $f(t) \leq f_0 + L \int^t_{t_0} |f(s)| \kD s \Rightarrow |f(t)| \leq f e^{L(t - t_0)}.$
}

\paragraph{自治系统}
\begin{equation}\label{Eq:1.25}
  \dot{x} = f(x),\  x(t_0) = x^0
\end{equation}
的解 $x = \Phi(t; t_0, x^0)$ 的最大存在区间 $J(x^0) = (\alpha, \beta)$, $t = t_0$, $x = x_0 \Leftrightarrow x^0 = \Phi(t_0; t_0, x^0)$.

$f(x)$ 称为向量场, $\mathbb{R}^n$ 称为相空间.

\begin{Prop}[关于 $t$ 平移不变性]\label{Pr:1.8}\
  \begin{enumerate}
    \item $\forall \tau$, $x = \Phi(t + \tau; t_0, x_0)$ 为方程 \eqref{Eq:1.25} 的解.
    \item $x = \Phi(t - t_0; 0, x_0)$ 满足 $x(t_0) = x^0$, $\Phi(t; t_0, x^0) = \Phi(t - t_0; 0, x^0)$.
  \end{enumerate}
\end{Prop}

\begin{Prop}[解唯一性]\label{Pr:1.9}\
  相空间过 $x^0$ 的轨线由 $x^0$ 唯一决定. (不同于存在唯一性.)
\end{Prop}
\begin{proof}
  设方程 \eqref{Eq:1.25} 有两条轨线, $\Phi_1(t; t_1, x_1)$ 与 $\Phi_2(t; t_2, x_2)$ 过 $x^0$: $\exists T_1$, $T_2$ 使得:
  \begin{equation*}
    \Phi_1(T_1; t_1, x_1) = x^0 = \Phi_2(T_2; t_2, x_2).
  \end{equation*}
  由性质 \ref{Pr:1.8} $\Phi_1(t + T_1 - T_2; t_1, x_1)$ 与 $\Phi(t; t_2, x_2)$ 均为解, 且满足 $x(T_2) = x^0$, 由解的存在唯一性有 $\Phi_1(t; t_1, x_1)$, $\Phi_2(t; t_2, x_2)$ 为相同轨线 (相空间).

\begin{Note}

  \begin{description}
    \item[$\therefore$] $\Phi(t; t_0, x^0)$ 可以写成 $\Phi(t, x^0)$;

    \item[再一步] $\Phi_t(x)$;

\begin{Prop}[恒同性]
  $\Phi_0(x) \equiv x$
\end{Prop}
\begin{proof}
  $\Phi_0(x) = \Phi_t(x)|_{t = 0} = \Phi(t, x)|_{t = 0} = \Phi(0, x) = x$.
\end{proof}

\begin{Prop}[可加性]
  $\Phi_{t_1 + t_2}(x) = \Phi_{t_1}\left(\Phi_{t_2}(x)\right)$.
\end{Prop}
\begin{proof}
  \begin{align*}
    \Phi_{t_1 + t_2}(x) & = \Phi(t_1 + t_2, x)\\
    & = \Phi\left(t_1, \Phi(t_2, x)\right)\\
    & = \Phi_{t_1}\left(\Phi_{t_2}(x)\right)
  \end{align*}
\end{proof}

\begin{Prop}[可交换]
  $\Phi_{t_1 + t_2} = \Phi_{t_1} \circ \Phi_{t_2}$.
\end{Prop}

\begin{Prop}[可逆]
  $\Phi_t \circ \Phi_ { - t} = \Phi_0$.
\end{Prop}

\begin{Def}
  称 $\{\Phi_t \mid t \in \mathbb{R}\}$ 为单参数连续变换群, 如果满足:
  \begin{inparaenum}[(i)]
    \item 恒同;
    \item 可加;
    \item 可交换.
  \end{inparaenum}
\end{Def}

\subsection{流 (连续, 离散) }

微分方程组
\begin{equation}
  \dot{x} = f(x),\ x \in G,
\end{equation}
与其定义的单参数交换群 $\{\Phi_t \mid t \in \mathbb{R}\}$, 称之为 $G$ 上的动力系统 (连续, 流).

连续 (可微) 映射
\begin{equation*}
  \Phi_t:\ G \to G,
\end{equation*}
满足恒同性、可加性、可交换, 则称之为 $G$ 上的拓扑 (可微) 动力系统.

\section{奇点, 闭轨线, 导算子}

\begin{Def}\
  \begin{enumerate}
    \item 微分方程 $\dot{x} = f(x)$, $x$ 是奇点、平衡点 $\Leftrightarrow$ 使 $f(x) = 0$ 的实零点.
    \item 迭代 $x' = g(x)$, $x$ 是不动点、平衡点 $\Leftrightarrow g(x) = x$.
  \end{enumerate}
\end{Def}

\begin{Exa}\
  \begin{enumerate}
    \item $\dot{x} = r x (1 - x)$, 平衡点: 0, 1.
    \item $x_{n + 1} = r x_n (1 - x_n)$, $n \leq 4$ 保证 $[0,1] \mapsto [0,1]$.
    \item $x_{n + 1} = x_n \exp[r (1 - x_n)]$ 这个保证了正性.
  \end{enumerate}
\end{Exa}

\begin{Note}$\ $
  \begin{enumerate}
    \item 正性: 初始 $ > 0$, 结果 $ > 0$.

        $\dot{x} = x r (1 - x)$, 对于任意 $\delta$, 在区间 $(0, \delta)$ 上有,
        \begin{align*}
          \int^\delta_0 \frac{\dot{x}}{x} & = \ln \frac{x(\delta)}{x(0)} = \int^\delta_0 (r (1 - x)) = \left. r x - \frac{r}{2} x^2\right|^\delta_0,\\
          x(\delta) & = x(0) \e^{r \delta - \frac{r}{2} \delta^2}.
        \end{align*}

    \item \begin{equation*}
            \frac{x(t + \Delta t) - x(t)}{\Delta t} = \frac{x(t) \exp(\Delta t [r (1 - x(t))] - 1)}{\Delta t},
          \end{equation*}
          \begin{itemize}
            \item $\Delta t = 1$, $t = n$, 离散;
            \item $\Delta t \to 0$, $\dot{x}(t) = x(t) r (1 - x(t))$.
          \end{itemize}
  \end{enumerate}
\end{Note}

  f(x) = a_0 + a_1 x + a_2 x^2 + \dots,
\end{equation*}
取一次近似:
\begin{equation*}
  f(x) \doteq a_0 + a_1 x,
\end{equation*}
则得到线性近似系统,
\begin{equation}\label{Eq:1.28}
  \dot{x} = \diff f(x_0) (x - x_0),
\end{equation}
其中
\begin{equation*}
  \diff f(x_0) =
  \left.\begin{pmatrix}
    f_1 x_1 & \cdots & f_1 x_n\\
    \vdots & \ddots & \vdots\\
    f_n x_1 & \cdots & f_n x_n
  \end{pmatrix}\right|_{x = x_0}.
\end{equation*}

结论: 若 $\diff f(x_0)$ 为双曲的 (特征值的实部 $\neq 0$) 则方程 \eqref{Eq:1.28} 决定方程 \eqref{Eq:1.27} (局部) 性态, 即局部同胚.

\paragraph{稳定性}\

渐进稳定 $\Leftrightarrow$ 邻域中的解趋于 $x^*$.

\begin{Def}\
  \begin{enumerate}
    \item $x^*$ 稳定: 初值在小区域, 轨道跑不出大区域. 见图 \ref{Fi:1.24}.

    $\forall U$, $\exists V$, s.t. $\forall n$, $g^n(V) \subset U$.

    \begin{figure}[!ht]
      \centering
      \input{Graph/T1_24_i.tex}
      \caption{$\forall x \in V \Rightarrow g^n(x) \in U$}\label{Fi:1.24}
    \end{figure}

    \item $x^*$ 吸引 (局部). 见图 \ref{Fi:1.25}.

    $\exists V$ s.t. $x \in V \Rightarrow \lim \limits_{n \to \infty} g^n(x) = x^*$.

    \begin{figure}[!ht]
      \centering
      \input{Graph/T1_25_i.tex}
      \caption{$\forall x \in V \Rightarrow \lim \limits_{n \to \infty} g^n(x) = x^*$.}\label{Fi:1.25}
    \end{figure}

    \item $\text{渐进稳定} = \text{稳定} + \text{吸引}$.

    \item 不稳定 $\Leftrightarrow$ 不是稳定的.
  \end{enumerate}
\end{Def}

\begin{Thm}[一维情况]
  \begin{equation*}
    \dot{x} = g(x),\ (g \in C^1),\ |g'(x^*)| < 1,
  \end{equation*}
  $\Rightarrow \exists V$, $\forall x \in V$, $\lim\limits_{n \to \infty} g^n(x) = x^*$.
\end{Thm}
\begin{proof}
  $\because g\in C^1$, $|g'(x^*)| < 1$, 令 $V = (x^* - \epsilon, x^* + \epsilon)$,

  $\Rightarrow \exists A < 1$, 使得 $|g'(x)| < A < 1$, ($x \in V$).

  中值定理
  \begin{align*}
    |g(x) - x^*| & = |g(x) - g(x^*)|,\\
    & = |g'(\delta) (x - x^*)|,\ (\delta \in V),\\
    & \leq A |x - x^*|,\\
    & < |x - x^*| < \epsilon.
  \end{align*}

  $\Rightarrow |g^n(x) - x^*| \leq A^n |x - x^*| \to 0$.

  $\therefore \lim\limits_{n \to \infty} g^n(x) = x^*$.
\end{proof}

\begin{Cor}
  \begin{equation*}
    |g'(x^*)| > 1 \Rightarrow \exists V = N(x^*),\ \forall x \in V \setminus \{x^*\},\ \exists n,\ \text{s.t.}\ g^n(x) \notin V.
  \end{equation*}
\end{Cor}
